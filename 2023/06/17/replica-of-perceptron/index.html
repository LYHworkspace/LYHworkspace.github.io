复本技巧

在贝叶斯最优框架下，写出配分函数 \[ \begin{equation} \mathcal{Z}(\mathbf{y}, \mathbf{X})=P(\mathbf{y},\mathbf{X}) = \int \mathrm{d}\mathbf{w}~ P(\mathbf{w}, \mathbf{y}, \mathbf{X})= \int \mathrm{d}\mathbf{w}~ P(\mathbf{y} | \mathbf{w}, \mathbf{X}) P(\mathbf{w}) \end{equation} \]

为了方便，引入变量 \(\mathbf{z}=\mathbf{w} \mathbf{X} / \sqrt{n}\)，因此有

\[ \begin{equation} P(\mathbf{y}|\mathbf{z}) = \int \mathrm{d}\mathbf{z}~ P(\mathbf{y} | \mathbf{w}, \mathbf{X})~ \delta\left( \mathbf{z} - \frac{1}{\sqrt{n}}\mathbf{w} \mathbf{X} \right) \end{equation} \] 配分函数改写为

\[ \begin{equation} \mathcal{Z}(\mathbf{y}, \mathbf{X}) = \int \mathrm{d}\mathbf{z}~ P(\mathbf{y} | \mathbf{z}) \int \mathrm{d}\mathbf{w}~ P(\mathbf{w}) ~ \delta\left( \mathbf{z} - \frac{1}{\sqrt{n}}\mathbf{w} \mathbf{X} \right) \end{equation} \]

为了求解自由能的淬火平均

\[ \begin{equation} \Phi = \frac{1}{n} \mathbb{E}_{\mathbf{y},\mathbf{X}}~ \log \mathcal{Z}(\mathbf{y}, \boldsymbol{X}) \end{equation} \]

使用复本方法将对 \(\log \mathcal{Z}(\mathbf{y}, \boldsymbol{X})\) 的平均转为对 \(\mathcal{Z}(\mathbf{y}, \boldsymbol{X})^r\) 的平均

\[ \begin{equation} \Phi = \frac{1}{n} \lim_{r\to 0} \frac{\partial \log \mathbb{E}_{\mathbf{y},\mathbf{X}}~ \left[ \mathcal{Z}(\mathbf{y}, \boldsymbol{X})^r \right]}{\partial r} \end{equation} \]

其中 \(r\) 是复本指标。下面求解复本平均 \(\mathbb{E}_{\mathbf{y},\mathbf{X}}~ \left[ \mathcal{Z}(\mathbf{y}, \boldsymbol{X})^r \right]\)

\[ \begin{equation} \begin{aligned} &\mathbb{E}_{\mathbf{y},\mathbf{X}}\left[\mathcal{Z}(\mathbf{y}, \mathbf{X})^r\right]\\ &= \mathbb{E}_{\mathbf{w}^\star,\mathbf{X}}\left[\prod_{a=1}^r \int_{\mathbb{R}^n} \mathrm{~d} \mathbf{z}^a P_{\mathrm{out}^a}\left(\mathbf{y} \mid \mathbf{z}^a\right) \int_{\mathbb{R}^d} \mathrm{~d} \mathbf{w}^a P_{\mathbf{w}^a}\left(\mathbf{w}^a\right) \delta\left(\mathbf{z}^a-\frac{1}{\sqrt{n}} \mathbf{X} \mathbf{w}^a\right)\right] \\ &= \mathbb{E}_{\mathbf{X}} \int_{\mathbb{R}^n} \mathrm{~d} \mathbf{y} \int_{\mathbb{R}^n} \mathrm{~d} \mathbf{z}^{\star} P_{\mathrm{out}^{\star}}\left(\mathbf{y} \mid \mathbf{z}^{\star}\right) \int_{\mathbb{R}^d} \mathrm{~d} \mathbf{w}^{\star} P_{\mathbf{w}^{\star}}\left(\mathbf{w}^{\star}\right) \delta\left(\mathbf{z}^{\star}-\frac{1}{\sqrt{n}} \mathbf{X} \mathbf{w}^{\star}\right) \\ & \times\left[\prod_{a=1}^r \int_{\mathbb{R}^n} \mathrm{~d} \mathbf{z}^a P_{\mathrm{out}^a}\left(\mathbf{y} \mid \mathbf{z}^a\right) \int_{\mathbb{R}^d} \mathrm{~d} \mathbf{w}^a P_{\mathbf{w}^a}\left(\mathbf{w}^a\right) \delta\left(\mathbf{z}^a-\frac{1}{\sqrt{n}} \mathbf{X} \mathbf{w}^a\right)\right] \\ &= \mathbb{E}_{\mathbf{X}} \int_{\mathbb{R}^n} \mathrm{~d} \mathbf{y} \prod_{a=0}^r \int_{\mathbb{R}^n} \mathrm{~d} \mathbf{z}^a P_{\mathrm{out}^a}\left(\mathbf{y} \mid \mathbf{z}^a\right) \int_{\mathbb{R}^d} \mathrm{~d} \mathbf{w}^a P_{\mathbf{w}^a}\left(\mathbf{w}^a\right) \delta\left(\mathbf{z}^a-\frac{1}{\sqrt{n}} \mathbf{X} \mathbf{w}^a\right) \end{aligned} \end{equation} \]

其中，对 \(y\) 的平均是由老师的权重 \(w^\star\) 带来的，将其设为第零个复本。假设所有的数据 \(\mathrm{X}\) 都是独立同分布的，根据中心极限定理，有

\[ \begin{equation} z_\mu^a= \frac{1}{\sqrt{n}} \sum_{i=1}^n x_i^{(\mu)} w_i^a \sim \mathcal{N}\left( \mathbb{E}_{\mathrm{X}}\left[z_\mu^a\right], \mathbb{E}_{\mathrm{X}}\left[z_\mu^a z_\mu^b\right] \right) \end{equation} \]

其中

\[ \begin{equation} \begin{aligned} & \mathbb{E}_{\mathrm{X}}\left[z_\mu^a\right]=\frac{1}{\sqrt{n}} \sum_{i=1}^n \mathbb{E}_{\mathrm{X}}\left[x_i^{(\mu)}\right] w_i^a=0 \\ & \mathbb{E}_{\mathrm{X}}\left[z_\mu^a z_\mu^b\right]=\frac{1}{n} \sum_{i j} \mathbb{E}_{\mathrm{X}}\left[x_i^{(\mu)} x_j^{(\mu)}\right] w_i^a w_j^b=\frac{1}{n} \sum_{i j} \delta_{i j} w_i^a w_j^b=\frac{1}{n} \mathbf{w}^a \cdot \mathbf{w}^b \end{aligned} \end{equation} \]

定义对称重叠矩阵（symmetric overlap matrix）

\[ \begin{equation} Q\left(\left\{\mathbf{w}^a\right\}\right) \equiv\left(\frac{1}{n} \mathbf{w}^a \cdot \mathbf{w}^b\right)_{a, b=0 . . r} \end{equation} \] 以及 $(z^a)_{a=0r} \(，\) i (w_i^a){a=0r}$，因此有

\[ \begin{equation} \tilde{\mathbf{z}}_\mu \sim P_{\tilde{\mathbf{z}}}(\tilde{\mathbf{z}} ; Q)=\mathcal{N}_{\tilde{\mathbf{z}}}\left(\mathbf{0}_{r+1}, Q\right) \qquad P_{\tilde{\mathrm{w}}}(\tilde{\mathbf{w}})=\prod_{a=0}^r P_{\mathrm{w}}\left(\tilde{w}^a\right) \end{equation} \]

配分函数可以化为

\[ \begin{equation} \begin{aligned} \mathbb{E}_{\mathbf{y}, \mathbf{X}}\left[\mathcal{Z}(\mathbf{y}, \mathbf{X})^r\right] & =\mathbb{E}_{\mathbf{X}} \int \mathrm{~d} \mathbf{y} \prod_{a=0}^r \int \mathrm{~d} \mathbf{z}^a P_{\mathrm{out}^a}\left(\mathbf{y} \mid \mathbf{z}^a\right) \int \mathrm{~d} \mathbf{w}^a P_{\mathbf{w}^a}\left(\mathbf{w}^a\right) \delta\left(\mathbf{z}^a-\frac{1}{\sqrt{d}} \mathbf{X} \mathbf{w}^a\right) \\ & =\left[\int \mathrm{d} y \int \mathrm{~d} \tilde{\mathbf{z}} P_{\text{out}}(y \mid \tilde{\mathbf{z}}) P_{\tilde{\mathbf{z}}}(\tilde{\mathbf{z}} ; Q(\tilde{\mathbf{w}}))\right]^d\left[\int \mathrm{~d} \tilde{\mathbf{w}} P_{\tilde{\mathbf{w}}}(\tilde{\mathbf{w}})\right]^n \end{aligned} \end{equation} \]

利用 \(\delta(x)\) 的傅里叶变换

\[ \begin{equation} \begin{aligned} 1 & =\int \mathrm{~d} Q \prod_{a \leq b} \delta\left(n Q_{a b}-\sum_{i=1}^n w_i^a w_i^b\right) \\ & \propto \int \mathrm{~d} Q \int \mathrm{~d} \hat{Q} \prod_{a \leq b} \exp\left\{ - \hat{Q}_{a b} \left(n Q_{a b}-\sum_{i=1}^n w_i^a w_i^b\right) \right\} \\ & \propto \int \mathrm{~d} Q \int \mathrm{~d} \hat{Q} \exp\left\{ -\sum_{a \leq b} \hat{Q}_{a b}\left(n Q_{a b}-\sum_{i=1}^n w_i^a w_i^b\right) \right\} \\ & \propto \int \mathrm{~d} Q \int \mathrm{~d} \hat{Q} \exp (-n \operatorname{Tr}[Q \hat{Q}]) \exp \left(\frac{1}{2} \sum_{i=1}^n \tilde{\mathbf{w}}_i^{\top} \hat{Q} \tilde{\mathbf{w}}_i\right) \end{aligned} \end{equation} \]

可以将复本平均转为对 \(Q\) 和 \(\hat{Q}\) 的积分

\[ \begin{equation} \begin{aligned} \mathbb{E}_{\mathbf{y}, \mathbf{X}}\left[\mathcal{Z}(\mathbf{y}, \mathbf{X})^r\right] \propto& \int \mathrm{~d} Q \int \mathrm{~d} \hat{Q}~ \exp (-n \operatorname{Tr}[Q \hat{Q}]) \exp \left(\frac{1}{2} \sum_{i=1}^n \tilde{\mathbf{w}}_i^{\top} \hat{Q} \tilde{\mathbf{w}}_i\right)\\ & \left[\int \mathrm{d} y \int \mathrm{~d} \tilde{\mathbf{z}}~ P_{\text {out }}(y \mid \tilde{\mathbf{z}}) P_{\tilde{\mathbf{z}}}(\tilde{\mathbf{z}} ; Q(\tilde{\mathbf{w}}))\right]^d \left[\int \mathrm{~d} \tilde{\mathbf{w}} P_{\tilde{\mathbf{w}}}(\tilde{\mathbf{w}})\right]^n \\ \propto & \iint \mathrm{~d} Q \mathrm{~d} \hat{Q} ~ e^{n \Phi^{(r)}(Q, \hat{Q})} \end{aligned} \end{equation} \]

其中，

\[ \begin{equation} \begin{aligned} &\Phi^{(r)}(Q, \hat{Q})=-\operatorname{Tr}[Q \hat{Q}]+\log \Psi_{\mathrm{w}}^{(r)}(\hat{Q})+\alpha \log \Psi_{\text {out }}^{(r)}(Q) \\ &\Psi_{\mathrm{w}}^{(r)}(\hat{Q})=\int \mathrm{~d} \tilde{\mathbf{w}}~ P_{\tilde{\mathbf{w}}}(\tilde{\mathbf{w}}) e^{\frac{1}{2} \tilde{\mathbf{w}} \hat{Q} \tilde{\mathbf{w}}} \\ &\Psi_{\text {out }}^{(r)}(Q)=\int \mathrm{d} y \int \mathrm{~d} \tilde{\mathbf{z}}~ P_{\tilde{z}}(\tilde{\mathbf{z}} ; Q) P_{\text {out }}(y \mid \tilde{\mathbf{z}}) \end{aligned} \end{equation} \]

以及，

\[ \begin{equation} P_{\tilde{z}}(\tilde{\mathbf{z}} ; Q)=\frac{e^{-\frac{1}{2} \tilde{\mathbf{z}}^{\mathrm{T}} Q^{-1} \tilde{\mathbf{z}}}}{\operatorname{det}(2 \pi Q)^{1 / 2}} \end{equation} \]

使用Laplace近似即可求解自由能

\[ \begin{equation} \Phi(\alpha)=\operatorname{extr}_{Q, \hat{Q}}\left\{\lim _{r \rightarrow 0} \frac{\partial \Phi^{(r)}(Q, \hat{Q})}{\partial r}\right\} \end{equation} \]

复本对称假设

接下来在复本对称假设下求解自由能的表达式。假设复本具有如下的对称形式 \[ \begin{equation} Q_{\mathrm{rs}}=\left(\begin{array}{cccc} Q^0 & m & \ldots & m \\ m & Q & \ldots & \ldots \\ \ldots & \ldots & \ldots & q \\ m & \ldots & q & Q \end{array}\right) \quad \hat{Q}_{\mathrm{rs}}=\left(\begin{array}{cccc} \hat{Q}^0 & \hat{m} & \ldots & \hat{m} \\ \hat{m} & -\frac{1}{2} \hat{Q} & \ldots & \ldots \\ \ldots & \ldots & \ldots & \hat{q} \\ \hat{m} & \ldots & \hat{q} & -\frac{1}{2} \hat{Q} \end{array}\right) \end{equation} \]

其中 \(Q^0 = \rho_{\mathbf{w}^\star} = \frac{1}{n} || \mathbf{w}^\star ||^2_2\) 是老师权重的overlap，\(m = \frac{1}{n} \mathbf{w}^a \cdot \mathbf{w}^\star\) 是老师与复本之间的overlap，\(Q = \frac{1}{n} || \mathbf{w}^a ||^2_2\) 是复本权重的overlap，\(q = \frac{1}{n} \mathbf{w}^a \cdot \mathbf{w}^b\) 是复本之间的overlap。

分别计算 \(\Phi^{(r)}(Q, \hat{Q})\) 中的三项

  • Trace term

\[ \begin{equation} \left.\operatorname{Tr}(Q \hat{Q})\right|_{\mathrm{rs}}=Q^0 \hat{Q}^0+r m \hat{m}-\frac{1}{2} r Q \hat{Q}+\frac{r(r-1)}{2} q \hat{q} \end{equation} \]

因此有

\[ \begin{equation} \begin{aligned} &\operatorname{extr} \left\{\lim _{r \rightarrow 0} \frac{\partial}{\partial r}\left( -\operatorname{Tr}[Q \hat{Q}] \right)\right\} \\ &= \operatorname{extr} \left\{\lim _{r \rightarrow 0} \frac{\partial}{\partial r} \left( -Q^0 \hat{Q}^0-r m \hat{m}+\frac{1}{2} r Q \hat{Q}-\frac{r(r-1)}{2} q \hat{q} \right)\right\} \\ &= \operatorname{extr} \left\{\lim _{r \rightarrow 0} \left(- m \hat{m}+\frac{1}{2} Q \hat{Q}-\left( r-\frac{1}{2} \right) q \hat{q} \right)\right\} \\ &= \operatorname{extr} \left\{ - m \hat{m}+\frac{1}{2} Q \hat{Q} +\frac{1}{2}q \hat{q} \right\} \end{aligned} \end{equation} \]

  • Prior integral

求解 \(\Psi_{\mathrm{w}}^{(r)}(\hat{Q})\) 的关键在于将矩阵乘法拆分 \[ \tilde{\mathbf{w}} \hat{Q}_{\mathrm{rs}} \tilde{\mathbf{w}} = w^\star \hat{Q}^0 w^\star + 2\sum_{a=1}^r w^\star \hat{m} w^a - \left( \hat{Q} + \hat{q} \right) \sum_{a=1}^r \left( w^a \right)^2 + \hat{q} \left( \sum_{a=1}^r w^a \right)^2 \]

可得 \[ \begin{equation} \begin{aligned} \left.\Psi_{\mathrm{w}}^{(r)}(\hat{Q})\right|_{\mathrm{rs}} & =\int \mathbf{d} \tilde{\mathbf{w}} P_{\tilde{\mathbf{w}}}(\tilde{\mathbf{w}}) e^{\frac{1}{2} \tilde{\mathbf{w}} \hat{Q}_{\mathrm{rs}} \tilde{\mathbf{w}}} \\ & =\mathbb{E}_{w^{\star}} e^{\frac{1}{2} \hat{Q}^0\left(w^{\star}\right)^2} \int \mathbf{d} \tilde{\mathbf{w}} P_{\tilde{\mathbf{w}}}(\tilde{\mathbf{w}}) e^{w^{\star} \hat{m} \sum_{a=1}^r w^a-\frac{1}{2}(\hat{Q}+\hat{q}) \sum_{a=1}^r\left(\tilde{w}^a\right)^2+\frac{1}{2} \hat{q}\left(\sum_{a=1}^r w^a\right)^2} \\ & =\mathbb{E}_{\xi, w^{\star}} e^{\frac{1}{2} \hat{Q}^0\left(w^{\star}\right)^2}\left[\mathbb{E}_w \exp \left(\left[\hat{m} w^{\star} w-\frac{1}{2}(\hat{Q}+\hat{q}) w^2+\hat{q}^{1 / 2} \xi w\right]\right)\right]^r \end{aligned} \end{equation} \]

因此 \[ \begin{equation} \begin{aligned} \Psi_{\mathrm{w}}(\hat{Q})&=\operatorname{extr} \left\{\lim _{r \rightarrow 0} \frac{\partial}{\partial r}\left( \log \Psi_{\mathrm{w}}^{(r)}(\hat{Q}) \right)\right\} \\ & =\operatorname{extr} \left\{\lim _{r \rightarrow 0} \frac{\partial}{\partial r} \mathbb{E}_{\xi, w^{\star}} {\color{red} r} \log \left[\mathbb{E}_w \exp \left(\hat{m} w^{\star} w-\frac{1}{2}(\hat{Q}+\hat{q}) w^2+\hat{q}^{1 / 2} \xi w \right)\right] \right\} \\ & = \operatorname{extr} \left\{ \mathbb{E}_{\xi, w^{\star}} \log \left[\mathbb{E}_w \exp \left(\hat{m} w^{\star} w-\frac{1}{2}(\hat{Q}+\hat{q}) w^2+\hat{q}^{1 / 2} \xi w\right)\right] \right\} \\ &= \operatorname{extr} \left\{ \mathbb{E}_{\xi, w^{\star}} \exp\left( -\frac{1}{2} \hat{q}^{-1}\hat{m}^2 (w^{\star})^2 + \xi \hat{q}^{-\frac{1}{2}} \hat{m} w^\star \right) \log \left[\mathbb{E}_w \exp \left(-\frac{1}{2}(\hat{Q}+\hat{q}) w^2+\hat{q}^{1 / 2} \xi w\right)\right] \right\} \end{aligned} \end{equation} \]

最后一步为了解耦老师与学生的权重，用了变量替换

\[ \xi \leftarrow \xi + \hat{q}^{-\frac{1}{2}} \hat{m} w^\star \]

  • Channel integral

求解 \(\Psi_{\text {out }}^{(r)}(Q)\) 的思路与求解 \(\Psi_{\mathrm{w}}^{(r)}(\hat{Q})\) 相同，难点在于求解 \(P_{\tilde{z}}(\tilde{\mathbf{z}} ; Q)\) 中的逆矩阵

\[ Q_{\mathrm{rs}}^{-1}=\left[\begin{array}{llll} Q_{00}^{-1} & Q_{01}^{-1} & Q_{01}^{-1} & Q_{01}^{-1} \\ Q_{01}^{-1} & Q_{11}^{-1} & Q_{12}^{-1} & Q_{12}^{-1} \\ Q_{01}^{-1} & Q_{12}^{-1} & Q_{11}^{-1} & Q_{12}^{-1} \\ Q_{01}^{-1} & Q_{12}^{-1} & Q_{12}^{-1} & Q_{11}^{-1} \end{array}\right] \] 其中， \[ \begin{equation} \begin{aligned} Q_{00}^{-1} =& \left(Q^0-r m(Q+(r-1) q)^{-1} m\right)^{-1} \\ Q_{01}^{-1} =& -\left(Q^0-r m(Q+(r-1) q)^{-1} m\right)^{-1} m(q+(r-1) q)^{-1} \\ Q_{11}^{-1} =& (Q-q)^{-1}-(Q+(r-1) q)^{-1} q(Q-q)^{-1} \\ &+(Q+(r-1) q)^{-1} m\left(Q^0-r m(Q+(r-1) q)^{-1} m\right)^{-1} m(Q+(r-1) q)^{-1} \\ Q_{12}^{-1} =& -(Q+(r-1) q)^{-1} q(Q-q)^{-1} \\ &+(Q+(r-1) q)^{-1} m\left(Q-r m(Q+(r-1) q)^{-1} m\right)^{-1} m(Q+(r-1) q)^{-1} \\ \operatorname{det} Q_{\mathrm{rs}} =& (Q-q)^{r-1}(Q+(r-1) q)\left(Q^0-r m(Q+(r-1) q)^{-1} m\right) \end{aligned} \end{equation} \]

因此有

\[ \begin{equation} \begin{aligned} \left.\Psi_{\text {out }}^{(r)}(Q)\right|_{\mathrm{rs}} & =\int \mathrm{d} y \int \mathrm{d} \tilde{\mathbf{z}} e^{-\frac{1}{2} \tilde{\mathbf{z}}^{\top} Q_{\mathrm{rs}}^{-1} \tilde{\mathbf{z}}-\frac{1}{2} \log \left(\operatorname{det}\left(2 \pi Q_{\mathrm{rs}}\right)\right)} P_{\mathrm{out}}(y \mid \tilde{\mathbf{z}}) \\ & =\mathbb{E}_{y, \xi} \mathrm{e}^{-\frac{1}{2} \log \left(\operatorname{det}\left(2 \pi Q_{\mathrm{rs}}\right)\right)} \\ & \times \int \mathrm{d} z^{\star} P_{\text {out }^{\star}}\left(y \mid z^{\star}\right) e^{-\frac{1}{2} Q_{00}^{-1}\left(z^{\star}\right)^2}\left[\int d z P_{\text {out }}(y \mid z) e^{-Q_{01}^{-1} z^{\star} z-\frac{1}{2}\left(Q_{11}^{-1}-Q_{12}^{-1}\right) z^2-Q_{12}^{-1 / 2} \xi z}\right]^r \end{aligned} \end{equation} \]

采用与求解 \(\Psi_{\mathrm{w}}^{(r)}(\hat{Q})\) 类似的处理方式，得到

\[ \Psi_{\text {out }}\left(Q, m, q ; \rho_{\mathrm{w}^{\star}}\right) \equiv \mathbb{E}_{y, \xi}\left[\mathcal{Z}_{\text {out }^{\star}}\left(y, m q^{-1 / 2} \xi, \rho_{\mathrm{w}^{\star}}-m q^{-1} m\right) \log \mathcal{Z}_{\text {out }}\left(y, q^{1 / 2} \xi, Q-q\right)\right] \]

 

将三项结果相加，最终得到 \[ \begin{equation} \Phi_{\mathrm{rs}}(\alpha) =\operatorname{extr}_{Q, \hat{Q}, q, \hat{q}, m, \hat{m}}\left\{-m \hat{m}+\frac{1}{2} Q \hat{Q}+\frac{1}{2} q \hat{q}+\Psi_{\mathbf{w}}(\hat{Q}, \hat{m}, \hat{q})+\alpha \Psi_{\text {out }}\left(Q, m, q ; \rho_{\mathrm{w}^{\star}}\right)\right\} \end{equation} \]

其中， \[ \begin{equation} \begin{aligned} &\Psi_{\mathrm{w}}(\hat{Q}, \hat{m}, \hat{q}) \equiv \mathbb{E}_{\xi}\left[\mathcal{Z}_{\mathrm{w}^{\star}}\left(\hat{m} \hat{q}^{-1 / 2} \xi, \hat{m} \hat{q}^{-1} \hat{m}\right) \log \mathcal{Z}_{\mathrm{w}}\left(\hat{q}^{1 / 2} \xi, \hat{Q}+\hat{q}\right)\right] \\ &\Psi_{\text {out }}\left(Q, m, q ; \rho_{\mathrm{w}^{\star}}\right) \equiv \mathbb{E}_{y, \xi}\left[\mathcal{Z}_{\mathrm{out}^{\star}}\left(y, m q^{-1 / 2} \xi, \rho_{\mathrm{w}^{\star}}-m q^{-1} m\right) \log \mathcal{Z}_{\mathrm{out}}\left(y, q^{1 / 2} \xi, Q-q\right)\right]\\ & \rho_{\mathbf{w}^{\star}}=Q^0=\lim _{n \rightarrow \infty} \mathbb{E}_{\mathbf{w}^{\star}} \frac{1}{n}\left\|\mathbf{w}^{\star}\right\|_2^2 \end{aligned} \end{equation} \]

在贝叶斯最优中，有 \[ \begin{equation} \mathcal{Z}_{\text {out }}=\mathcal{Z}_{\text {out }^{\star}} \qquad \mathcal{Z}_{\mathbf{w}}=\mathcal{Z}_{\mathbf{w}^{\star}} \end{equation} \]

并且根据 Nishimori conditions 有 \[ \begin{equation} Q=\rho_{\mathrm{w}^{\star}},\quad m=q=q,\quad \hat{Q}=0, \quad \hat{m}=\hat{q}=\hat{q} \end{equation} \]

复本对称自由能简化为

\[ \begin{equation} \Phi(\alpha)=\operatorname{extr}_{q, \hat{q}}\left\{-\frac{1}{2} q \hat{q}+\Psi_{\mathrm{w}}^{\mathrm{b}}\left(\hat{q}\right)+\alpha \Psi_{\text {out }}^{\mathrm{b}}\left(q ; \rho_{\mathrm{w}^{\star}}\right)\right\} \end{equation} \]

其中， \[ \begin{equation} \begin{aligned} & \Psi_{\mathrm{w}}(\hat{q}) =\mathbb{E}_{\xi}\left[\mathcal{Z}_{\mathrm{w}^{\star}}\left(\hat{q}^{1 / 2} \xi, \hat{q}\right) \log \mathcal{Z}_{\mathrm{w}^{\star}}\left(\hat{q}^{1 / 2} \xi, \hat{q}\right)\right], \\ & \Psi_{\text {out }}\left(q ; \rho_{\mathrm{w}^{\star}}\right)=\mathbb{E}_{y, \xi}\left[\mathcal{Z}_{\text {out }^{\star}}\left(y, q^{1 / 2} \xi, \rho_{\mathrm{w}^{\star}}-q\right) \log \mathcal{Z}_{\text {out }^{\star}}\left(y, q^{1 / 2} \xi, \rho_{\mathrm{w}^{\star}}-q\right)\right] \end{aligned} \end{equation} \]

通过

\[ \begin{equation} \begin{array}{rlrl} \hat{Q} & =-2 \alpha \partial_Q \Psi_{\text {out }}, && Q=-2 \partial_{\hat{Q}} \Psi_{\mathrm{w}} \\ \hat{q} & =-2 \alpha \partial_q \Psi_{\text {out }}, && q=-2 \partial_{\hat{q}} \Psi_{\mathrm{w}} \\ \hat{m} & =\alpha \partial_m \Psi_{\text {out }}, & &m=\partial_{\hat{m}} \Psi_{\mathrm{w}} \end{array} \end{equation} \]

可以分别求得各个序参量，但是在本问题中，根据 Nishimori conditions，只需要求出 \(q\) 和 \(\hat{q}\)。

在求偏导时定义辅助变量

\[ \begin{aligned} & f_{\mathrm{w}^{\star}}(\gamma, \Lambda) \equiv \partial_\gamma \log \mathcal{Z}_{\mathrm{w}^{\star}}(\gamma, \Lambda)=\frac{\gamma}{1+\Lambda} \\ & f_{\text {out }^{\star}}(y, \omega, V) \equiv \partial_\omega \log \mathcal{Z}_{\text {out }^{\star}}(y, \omega, V)=\frac{\mathcal{N}_y\left(1, \Delta^{\star}\right)-\mathcal{N}_y\left(-1, \Delta^{\star}\right)}{\mathcal{Z}_{\text {out }\star} (y, \omega, V)} \mathcal{N}_\omega(0, V) \end{aligned} \]

可得（计算过程待以后补充）

\[ \begin{equation} \begin{aligned} & \hat{q}=\alpha \mathbb{E}_{y, \xi}\left[\mathcal{Z}_{\mathrm{out}^{\star}}\left(y, q^{1 / 2} \xi, \rho_{\mathrm{w}^{\star}}-q\right) f_{\mathrm{out}^{\star}}\left(y, q^{1 / 2} \xi, \rho_{\mathrm{w}^{\star}}-q\right)^2\right] \\ & q=\mathbb{E}_{\xi}\left[\mathcal{Z}_{\mathrm{w}^{\star}}\left(\hat{q}^{1 / 2} \xi, \hat{q}\right) f_{\mathrm{w}^{\star}}\left(\hat{q}^{1 / 2} \xi, \hat{q}\right)^2\right] \end{aligned} \end{equation} \]

为了给出 \(\mathcal{Z}_{\mathrm{out}}\) 的解析形式，我们用一个高斯分布取极限的形式来代替 \(\delta\) 函数

\[ \begin{equation} P(y \mid z)=\frac{1}{\sqrt{2 \pi \Delta}} \exp \left(\frac{y-\operatorname{sign}(z)}{2 \Delta}\right), \Delta \rightarrow \infty \end{equation} \]

因此

\[ \begin{equation} \begin{aligned} \mathcal{Z}_{\text {out }}(y, \omega, V)= & \int d z \frac{1}{\sqrt{2 \pi \Delta}} \exp \left(\frac{y-\operatorname{sign}(z)}{2 \Delta}\right) \frac{1}{\sqrt{2 \pi V}} \exp \left(-\frac{(z-\omega)^2}{2 V}\right) \\ = & \int_0^{\infty} d z \frac{1}{\sqrt{2 \pi \Delta}} \exp \left(\frac{y-1}{2 \Delta}\right) \frac{1}{\sqrt{2 \pi V}} \exp \left(-\frac{(z-\omega)^2}{2 V}\right) \\ + & \int_{-\infty}^0 d z \frac{1}{\sqrt{2 \pi \Delta}} \exp \left(\frac{y+1}{2 \Delta}\right) \frac{1}{\sqrt{2 \pi V}} \exp \left(-\frac{(z-\omega)^2}{2 V}\right) \\ = & \mathcal{N}_y\left(1, \Delta^*\right) \left(1+\operatorname{erf}\left(\frac{\omega}{\sqrt{2 V}}\right)\right)+\mathcal{N}_y\left(-1, \Delta^*\right) \frac{1}{2}\left(1-\operatorname{erf}\left(\frac{\omega}{\sqrt{2 V}}\right)\right) \\ f_{\text {out }^{\star}}(y, \omega, V) =& \frac{\mathcal{N}_y\left(1, \Delta^{\star}\right)-\mathcal{N}_y\left(-1, \Delta^{\star}\right)}{\mathcal{Z}_{\text {out }^{\star}}(y, \omega, V)} \mathcal{N}_\omega(0, V) \end{aligned} \end{equation} \]

同样的，在 \(P_{\mathbf{w}^\star} \sim \mathcal{N}(0,1)\) 时，有 \[ \begin{equation} \mathcal{Z}_{\mathrm{w}^{\star}}(\gamma, \Lambda)=\frac{e^{\frac{\gamma^2}{2(\Lambda +1)}}}{\sqrt{\Lambda +1}}, \quad f_{\mathrm{w}^{\star}}(\gamma, \Lambda)=\frac{\gamma}{1+\Lambda}, \quad \partial_\gamma f_{\mathrm{w}^{\star}}(\gamma, \Lambda)=\frac{1}{1+\Lambda} \end{equation} \]

因此

\[ \begin{equation} \begin{aligned} q &=\frac{\hat{q}}{1+\hat{q}} \\ \hat{q} & =\alpha \mathbb{E}_{y, \xi}\left[\mathcal{Z}_{\text {out }^{\star}}\left(y, q^{1 / 2} \xi, 1-q\right) f_{\text {out }^{\star}}\left(y, q^{1 / 2} \xi, 1-q\right)^2\right] \\ & =2 \alpha \int D \xi y^2 \frac{\mathcal{N}_{\sqrt{q} \xi}\left(0,1-q\right)^2}{\frac{1}{2}\left(1+\operatorname{erf}\left(\frac{\sqrt{q \xi}}{\sqrt{2\left(1-q\right)}}\right)\right)}\\ &=\frac{2}{\pi} \frac{\alpha}{1-q} \int D \xi \frac{e^{-\frac{q_{\mathrm{b} \xi^2}}{1-q}}}{\left(1+\operatorname{erf}\left(\frac{\sqrt{q \xi}}{\sqrt{2\left(1-q\right)}}\right)\right)} \end{aligned} \end{equation} \]
